meshed.makers.edge_reversed_graph(g: Mapping[T, Iterable[T]], dst_nodes_factory: Callable[[], Iterable[T]] = <class 'list'>, dst_nodes_append: Callable[[Iterable[T], T], None] = <method 'append' of 'list' objects>) → Mapping[T, Iterable[T]][source]¶
>>> g = dict(a='c', b='cd', c='abd', e='')
>>> assert edge_reversed_graph(g) == {'c': ['a', 'b'], 'd': ['b', 'c'], 'a': ['c'], 'b': ['c'], 'e': []}
>>> reverse_g_with_sets = edge_reversed_graph(g, set, set.add)
>>> assert reverse_g_with_sets == {'c': {'a', 'b'}, 'd': {'b', 'c'}, 'a': {'c'}, 'b': {'c'}, 'e': set([])}
